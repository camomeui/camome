## Usage

```tsx live
import { FormControl, TextInput } from "@camome/components";

export default function () {
  return (
    <FormControl labelText="Name">
      <TextInput required placeholder="John Doe" />
    </FormControl>
  );
}
```

## States

### Error

```tsx live
import { FormControl, TextInput } from "@camome/components";

export default function () {
  return (
    <FormControl
      labelText="Name"
      helperText="Tell us your identifier"
      error="Something is wrong"
    >
      <TextInput required placeholder="John Doe" />
    </FormControl>
  );
}
```

### Disabled

```tsx live
import { TextInput } from "@camome/components";

export default function () {
  return <TextInput disabled placeholder="You can't type..." />;
}
```

## Sizes

```tsx live
import { TextInput } from "@camome/components";

export default function () {
  return (
    <>
      <TextInput size="sm" placeholder="Small" />
      <TextInput size="md" placeholder="Medium" />
      <TextInput size="lg" placeholder="Large" />
    </>
  );
}
```
