## Usage

```tsx live
import { FormControl, Select } from "@camome/components";

export default function () {
  return (
    <FormControl labelText="Your position">
      <Select>
        <option value="developer">Developer</option>
        <option value="designer">Designer</option>
        <option value="marketer">Marketer</option>
      </Select>
    </FormControl>
  );
}
```

## States

### Error

```tsx live
import { FormControl, Select } from "@camome/components";

export default function () {
  return (
    <FormControl labelText="Your position" error="Something is wrong">
      <Select>
        <option value="developer">Developer</option>
        <option value="designer">Designer</option>
        <option value="marketer">Marketer</option>
      </Select>
    </FormControl>
  );
}
```

### Disabled

```tsx live
import { Select } from "@camome/components";

export default function () {
  return (
    <Select disabled>
      <option value="developer">Developer</option>
    </Select>
  );
}
```
