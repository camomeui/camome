## Usage

```tsx live
import { Checkbox } from "@camome/components";

export default function () {
  return (
    <Checkbox labelText="Agree to our Privacy Policy" value="true" required />
  );
}
```

## Sizes

```tsx live
import { Checkbox } from "@camome/components";

export default function () {
  return (
    <>
      <Checkbox
        labelText="Agree to our Privacy Policy"
        value="true"
        size="sm"
      />
      <Checkbox
        labelText="Agree to our Privacy Policy"
        value="true"
        size="md"
      />
      <Checkbox
        labelText="Agree to our Privacy Policy"
        value="true"
        size="lg"
      />
    </>
  );
}
```

## States

### Disabled

```tsx live
import { Checkbox } from "@camome/components";

export default function () {
  return (
    <>
      <Checkbox labelText="Checked" value="true" checked disabled />
      <Checkbox labelText="Not checked" value="true" disabled />
    </>
  );
}
```

## With helper text

```tsx live
import { Checkbox } from "@camome/components";

export default function () {
  return (
    <Checkbox
      labelText="With helper text"
      helperText="Descriptive text"
      value="true"
    />
  );
}
```
