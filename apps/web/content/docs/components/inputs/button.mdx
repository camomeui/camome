## Usage

```tsx live
import { Button } from "@camome/components";

export default function Example() {
  return <Button variant="solid">Click me!</Button>;
}
```

## Variants

```tsx live
import { Button } from "@camome/components";

export default function Example() {
  return (
    <>
      <Button variant="solid">Solid</Button>
      <Button variant="subtle">Subtle</Button>
      <Button variant="outline">Outline</Button>
      <Button variant="ghost">Ghost</Button>
    </>
  );
}
```

## Color schemes

```tsx live
import { Button } from "@camome/components";

export default function Example() {
  return (
    <>
      <Button colorScheme="primary">Primary</Button>
      <Button colorScheme="secondary">Secondary</Button>
      <Button colorScheme="info">Info</Button>
      <Button colorScheme="success">Success</Button>
      <Button colorScheme="warn">Warn</Button>
      <Button colorScheme="danger">Danger</Button>
    </>
  );
}
```

## States

```tsx live
import { Button, Spinner } from "@camome/components";

export default function Example() {
  return (
    <>
      <Button variant="solid" disabled>
        Disabled
      </Button>
      <Button leftIcon={<Spinner />} disabled>
        Processing...
      </Button>
    </>
  );
}
```

## Sizes

```tsx live
import { Button } from "@camome/components";

export default function Example() {
  return (
    <>
      <Button size="sm">Small</Button>

}
```
