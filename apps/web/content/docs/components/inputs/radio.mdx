## RadioGroup

```tsx live
import { FormControl, RadioGroup, Radio } from "@camome/components";

export default function () {
  return (
    <FormControl labelText="Gundam kids">
      <RadioGroup aria-required>
        <Radio
          labelText="Katz"
          name="gundam-kids-1"
          value="katz"
          defaultChecked
        />
        <Radio labelText="Letz" name="gundam-kids-1" value="letz" />
        <Radio labelText="Kikka" name="gundam-kids-1" value="kikka" />
      </RadioGroup>
    </FormControl>
  );
}
```
