---
title: "Radio"
id: "radio"
description: "Radio buttons allow the user to select one option from a set."
---

## Usage

```tsx live
<FormControl labelText="Gundam kids">
  <RadioGroup aria-required>
    <Radio labelText="Katz" name="gundam-kids-1" value="katz" defaultChecked />
    <Radio labelText="Letz" name="gundam-kids-1" value="letz" />
    <Radio labelText="Kikka" name="gundam-kids-1" value="kikka" />
  </RadioGroup>
</FormControl>;
```

## Sizes

```tsx live
<FormControl labelText="Gundam kids">
  <RadioGroup aria-required>
    <Radio
      labelText="Katz"
      name="gundam-kids-2"
      value="katz"
      size="sm"
      defaultChecked
    />
    <Radio labelText="Letz" name="gundam-kids-2" value="letz" size="md" />
    <Radio labelText="Kikka" name="gundam-kids-2" value="kikka" size="lg" />
  </RadioGroup>
</FormControl>;
```

## Horizontal RadioGroup

```tsx live
<FormControl labelText="Gundam kids">
  <RadioGroup aria-required direction="horizontal">
    <Radio labelText="Katz" name="gundam-kids-3" value="katz" defaultChecked />
    <Radio labelText="Letz" name="gundam-kids-3" value="letz" />
    <Radio labelText="Kikka" name="gundam-kids-3" value="kikka" />
  </RadioGroup>
</FormControl>;
```

## States

### Disabled

```tsx live
<>
  <Radio labelText="Checked" value="true" checked disabled />
  <Radio labelText="Not checked" value="true" disabled />
</>;
```

## With helper text

```tsx live
<FormControl
  labelText="Gundam kids"
  helperText="Choose your favorite"
  error="This field is required"
>
  <RadioGroup aria-required>
    <Radio labelText="Katz" name="gundam-kids-4" value="katz" />
    <Radio labelText="Letz" name="gundam-kids-4" value="letz" />
    <Radio labelText="Kikka" name="gundam-kids-4" value="kikka" />
  </RadioGroup>
</FormControl>;
```
