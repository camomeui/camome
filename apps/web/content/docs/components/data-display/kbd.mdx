## Sizes

Just set the `font-size` of the parent to change the size.

```tsx live
import { Kbd } from "@camome/components";

export default function () {
  return (
    <span
      style={{
        display: "flex",
        alignItems: "baseline",
        gap: "0.3rem",
        fontSize: "var(--cmm-font-size-md)",
      }}
    >
      <Kbd>Ctrl</Kbd>+<Kbd>Alt</Kbd>+<Kbd>Delete</Kbd>
    </span>
  );
}
```
