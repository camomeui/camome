## Overview

```tsx live
import { Avatar } from "@camome/components";

export default function () {
  return (
    <>
      <Avatar src="https://i.imgur.com/isPfQ2E.jpg" alt="Alt Name" />
      <Avatar variant="subtle" colorScheme="primary">
        JD
      </Avatar>
      <Avatar variant="solid">EH</Avatar>
      <Avatar colorScheme="success" />
    </>
  );
}
```

## With custom content

```tsx live
import { Avatar } from "@camome/components";
import { StarIcon } from "@heroicons/react/24/outline";

export default function () {
  return (
    <Avatar>
      <StarIcon />
    </Avatar>
  );
}
```

## As clickable item

```tsx live
import { Avatar } from "@camome/components";

export default function () {
  return (
    <>
      <Avatar
        src="https://i.imgur.com/FEnPQp7.jpg"
        alt="Alt Name"
        component="a"
        href="#"
      />
      <Avatar
        src="https://i.imgur.com/QTo2MPU.jpg"
        alt="Alt Name"
        component="button"
        onClick={() => alert("You clicked successfully")}
      />
    </>
  );
}
```
