description: "A Breadcrumbs trail enables navigation to ascendant routes of the current page."
---

## Usage

```tsx live
import { Breadcrumbs } from "@camome/components";

export default function () {
  return (
    <Breadcrumbs
      paths={[
        {
          href: "#home",
          label: "Home",
        },
        {
          href: "#category",
          label: "Category",
        },
        {
          label: "Product",
        },
      ]}
    />
  );
}
```
