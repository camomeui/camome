## Usage with Headless UI

```tsx live
function Example() {
  const category1 = [
    {
      label: "Edit",
      icon: <PencilSquareIcon />,
    },
    {
      label: "Duplicate",
      icon: <Square2StackIcon />,
    },
  ];

  const category2 = [
    {
      label: "Delete",
      icon: <TrashIcon />,
      "--font-color": "var(--cmm-color-danger-600)",
      "--icon-color": "var(--cmm-color-danger-600)",
      "--bg-hover": "var(--cmm-color-danger-50)",
    },
  ];

  const { useFloating, offset, flip } = Floating;
  const { x, y, reference, floating, strategy } = useFloating({
    placement: "bottom",
    middleware: [offset(8), flip()],
  });
  const { menu, item, itemActive, label, divider } = menuClassNames;

  return (
    <HeadlessMenu as="div">
      <HeadlessMenu.Button as="span">
        {(open) => <Button ref={reference}>Open</Button>}
      </HeadlessMenu.Button>
      <HeadlessMenu.Items
        className={menu}
        ref={floating}
        style={{
          position: strategy,
          top: y ?? 0,
          left: x ?? 0,
          width: "max-content",
        }}
      >
        <Menu.Label>Editing</Menu.Label>
        {category1.map(({ label, icon, ...style }) => (
          <HeadlessMenu.Item
            key={label}
            as="button"
            className={({ active }) => clsx(item, active && itemActive)}
            style={style}
          >
            {icon}
            {label}
          </HeadlessMenu.Item>
        ))}
        <Menu.Divider />
        <Menu.Label>Danger</Menu.Label>
        {category2.map(({ label, icon, ...style }) => (
          <HeadlessMenu.Item
            key={label}
            as="button"
            className={({ active }) => clsx(item, active && itemActive)}
            style={style}
          >
            {icon}
            {label}
          </HeadlessMenu.Item>
        ))}
      </HeadlessMenu.Items>
    </HeadlessMenu>
  );
}
```

## Style-only components

```tsx live
function Example() {
  const category1 = [
    {
      label: "Edit",
      icon: <PencilSquareIcon />,
    },
    {
      label: "Duplicate",
      icon: <Square2StackIcon />,
    },
  ];

  const category2 = [
    {
      label: "Delete",
      icon: <TrashIcon />,
      "--font-color": "var(--cmm-color-danger-600)",
      "--icon-color": "var(--cmm-color-danger-600)",
      "--bg-hover": "var(--cmm-color-danger-50)",
    },
  ];

  return (
    <Menu>
      <Menu.Label>Editing</Menu.Label>
      {category1.map((item) => (
        <Menu.Item key={item.label} component="a">
          {item.icon}
          {item.label}
        </Menu.Item>
      ))}
      <Menu.Divider />
      <Menu.Label>Danger</Menu.Label>
      {category2.map(({ label, icon, ...style }) => (
        <Menu.Item key={label} style={style}>
          {icon}
          {label}
        </Menu.Item>
      ))}
    </Menu>
  );
}
```
